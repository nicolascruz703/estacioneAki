alter table logParquimetro  modify dataHoraOcupacao varchar (19);
alter table logParquimetro  modify placaVeiculoOcupacao varchar (8);
alter table logParquimetro  modify tempoOcupacao varchar (10);
alter table logParquimetro  modify valorAPagar varchar (10);
alter table logParquimetro  modify valorDevido varchar (10);

DELIMITER $$
CREATE TRIGGER triggerLogParquimetro 
AFTER UPDATE ON Parquimetro
FOR EACH ROW 
BEGIN
insert into logParquimetro (dataHoraOcupacao, idParquimetro, idVaga, placaVeiculoOcupacao, tempoOcupacao, valorAPagar, valorDevido) select dataHoraOcupacao, id, idVaga, placaVeiculoOcupacao, tempoOcupacao, concat('R$', valorAPagar,',00'), concat('R$', valorAPagar-valorPago,',00') from parquimetro where statusVaga=3;
END$$
DELIMITER ;
